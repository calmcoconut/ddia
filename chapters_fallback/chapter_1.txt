    {
      "term": "Online Transaction Processing (OLTP)",
      "definition": "Interactive database workloads characterized by low-latency, high-volume point queries, and real-time creations, updates, or deletions of individual records corresponding to current business operations."
    },
    {
      "term": "Online Analytical Processing (OLAP)",
      "definition": "Database workloads characterized by low-volume, highly complex queries that scan and aggregate massive historical datasets (terabytes to petabytes) to support business intelligence and strategic decision-making."
    },
    {
      "term": "Extract-Transform-Load (ETL) / Extract-Load-Transform (ELT)",
      "definition": "Data pipeline processes used to copy, clean, restructure, and load transactional data from multiple operational source databases (or SaaS APIs) into a centralized analytical data warehouse."
    },
    {
      "term": "Hybrid Transactional/Analytical Processing (HTAP)",
      "definition": "Unified database architectures designed to support concurrent transactional modifications (OLTP) and large-scale analytical scans (OLAP) simultaneously without requiring external ETL synchronization."
    },
    {
      "term": "Data Warehouse",
      "definition": "A centralized, read-only relational database optimized for executing complex analytical queries and aggregations over historical, schema-conforming operational data."
    },
    {
      "term": "Data Lake",
      "definition": "A flexible, centralized storage repository that holds a copy of raw, unstructured, or semi-structured data files (such as Parquet or Avro) without imposing an upfront data model or schema, serving the needs of data scientists."
    },
    {
      "term": "System of Record (Source of Truth)",
      "definition": "The authoritative database that holds the canonical, primary version of a piece of data. Writes must be directed here first, and its representation is typically normalized to represent each fact exactly once."
    },
    {
      "term": "Derived Data System",
      "definition": "A database, index, cache, or materialized view whose content is computed from existing data in a system of record. Although redundant, derived systems are critical for accelerating complex read queries and can be entirely reconstructed from the source of truth if lost."
    },
    {
      "term": "Disaggregation (Separation of Storage and Compute)",
      "definition": "An architectural paradigm, common in cloud-native systems, that decouples the computing layer (CPU/RAM instances) from the persistent storage layer (e.g., managed object storage like Amazon S3) so they can scale independently."
    },
    {
      "term": "Multitenancy",
      "definition": "A software architecture where multiple distinct customers or applications share the same physical server hardware and runtime instances while relying on the software to isolate their data and resource usage."
    },
    {
      "term": "Distributed System",

    },
    {
